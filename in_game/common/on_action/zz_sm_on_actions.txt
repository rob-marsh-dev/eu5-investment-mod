# The Stock Exchange — monthly market pulse.
# Appends to the base game's monthly_country_pulse via the additive on_actions
# list (lists merge across mods — same hook the banking mod uses, no conflict).
monthly_country_pulse = {
	on_actions = { sm_market_monthly_pulse }
}

# root = the country being pulsed (the market/target country).
# 1) Drift this country's market sentiment (random walk, clamped 0.7–1.3) so
#    prices boom and slump even in peacetime.
# 2) Cache all valuations (the per-frame GUI may only read variables, and the
#    dividend payouts read the same cache).
# 3) Pay dividends out of THIS country's treasury to each concession holder.
#    No safety rails: a heavily sold-off country genuinely bleeds, and
#    dividends can push its treasury negative.
sm_market_monthly_pulse = {
	effect = {
		# --- Sentiment random walk ---
		if = {
			limit = { NOT = { exists = var:sm_market_sentiment } }
			set_variable = { name = sm_market_sentiment value = 1 }
		}
		random_list = {
			30 = { change_variable = { name = sm_market_sentiment add = 0.04 } }
			40 = { }
			30 = { change_variable = { name = sm_market_sentiment subtract = 0.04 } }
		}
		if = {
			limit = { var:sm_market_sentiment > 1.3 }
			set_variable = { name = sm_market_sentiment value = 1.3 }
		}
		if = {
			limit = { var:sm_market_sentiment < 0.7 }
			set_variable = { name = sm_market_sentiment value = 0.7 }
		}

		# --- Cache valuations (GUI display + this month's dividends) ---
		# Runs BEFORE the payouts: the dividend script values read
		# sm_X_paying_book from this cache, so the locations are walked
		# once per sector per month, here and nowhere else. Per-frame GUI
		# ScriptValues also read only these variables.
		#   sm_X_book        full book value (drives prices — untouched
		#                    by occupation; Rob wants prices kept high)
		#   sm_X_paying_book book of the locations the target still
		#                    CONTROLS — the dividend base. Occupied land
		#                    pays its concession holder nothing.
		#   sm_X_value       market value (book x situation x sentiment)
		# Commerce has no paying cache: its book is income-based and
		# already collapses under occupation, suspending its dividend.
		set_variable = { name = sm_agriculture_book value = sm_agriculture_book_value }
		set_variable = { name = sm_agriculture_paying_book value = sm_agriculture_paying_book_value }
		set_variable = {
			name = sm_agriculture_value
			value = {
				value = var:sm_agriculture_book
				multiply = sm_market_situation_mult
				multiply = sm_market_sentiment_mult
			}
		}
		set_variable = { name = sm_mining_book value = sm_mining_book_value }
		set_variable = { name = sm_mining_paying_book value = sm_mining_paying_book_value }
		set_variable = {
			name = sm_mining_value
			value = {
				value = var:sm_mining_book
				multiply = sm_market_situation_mult
				multiply = sm_market_sentiment_mult
			}
		}
		set_variable = { name = sm_industry_book value = sm_industry_book_value }
		set_variable = { name = sm_industry_paying_book value = sm_industry_paying_book_value }
		set_variable = {
			name = sm_industry_value
			value = {
				value = var:sm_industry_book
				multiply = sm_market_situation_mult
				multiply = sm_market_sentiment_mult
			}
		}
		set_variable = { name = sm_commerce_book value = sm_commerce_book_value }
		set_variable = {
			name = sm_commerce_value
			value = {
				value = var:sm_commerce_book
				multiply = sm_market_situation_mult
				multiply = sm_market_sentiment_mult
			}
		}

		# --- Dividends: treasury -> concession holder, sector by sector ---
		# Paid on the just-cached PAYING book (occupied locations pay
		# nothing — dividend suspension). Each payout also self-heals the
		# holder's sm_investment_targets list (the Investments-window
		# datamodel): stakes acquired before the list existed, or missed
		# by any future acquisition path, get their ledger row back
		# within a month.
		if = {
			limit = { exists = var:sm_agriculture_stake_owner }
			var:sm_agriculture_stake_owner = {
				add_gold = root.sm_agriculture_dividend
				if = {
					limit = {
						NOT = {
							is_target_in_variable_list = {
								name = sm_investment_targets
								target = root
							}
						}
					}
					add_to_variable_list = { name = sm_investment_targets target = root }
				}
			}
			add_gold = {
				value = sm_agriculture_dividend
				multiply = -1
			}
		}
		if = {
			limit = { exists = var:sm_mining_stake_owner }
			var:sm_mining_stake_owner = {
				add_gold = root.sm_mining_dividend
				if = {
					limit = {
						NOT = {
							is_target_in_variable_list = {
								name = sm_investment_targets
								target = root
							}
						}
					}
					add_to_variable_list = { name = sm_investment_targets target = root }
				}
			}
			add_gold = {
				value = sm_mining_dividend
				multiply = -1
			}
		}
		if = {
			limit = { exists = var:sm_industry_stake_owner }
			var:sm_industry_stake_owner = {
				add_gold = root.sm_industry_dividend
				if = {
					limit = {
						NOT = {
							is_target_in_variable_list = {
								name = sm_investment_targets
								target = root
							}
						}
					}
					add_to_variable_list = { name = sm_investment_targets target = root }
				}
			}
			add_gold = {
				value = sm_industry_dividend
				multiply = -1
			}
		}
		if = {
			limit = { exists = var:sm_commerce_stake_owner }
			var:sm_commerce_stake_owner = {
				add_gold = root.sm_commerce_dividend
				if = {
					limit = {
						NOT = {
							is_target_in_variable_list = {
								name = sm_investment_targets
								target = root
							}
						}
					}
					add_to_variable_list = { name = sm_investment_targets target = root }
				}
			}
			add_gold = {
				value = sm_commerce_dividend
				multiply = -1
			}
		}

		# --- Nationalization temper roll (Phase 3) ---
		# The hostile end of the "get the bank out" scale: one roll per
		# invested country per month. Tiers are exclusive (first match
		# wins) and all read the same cheap inputs — an opinion trigger
		# and the treasury. Content, solvent countries never roll at all.
		# Costs one OR of exists-checks for the 99% of countries with no
		# foreign concession. SP mod: in practice one bank per country,
		# so the first set owner is THE bank.
		if = {
			limit = {
				OR = {
					exists = var:sm_agriculture_stake_owner
					exists = var:sm_mining_stake_owner
					exists = var:sm_industry_stake_owner
					exists = var:sm_commerce_stake_owner
				}
			}
			if = {
				limit = { exists = var:sm_agriculture_stake_owner }
				var:sm_agriculture_stake_owner = { save_temporary_scope_as = sm_bank }
			}
			else_if = {
				limit = { exists = var:sm_mining_stake_owner }
				var:sm_mining_stake_owner = { save_temporary_scope_as = sm_bank }
			}
			else_if = {
				limit = { exists = var:sm_industry_stake_owner }
				var:sm_industry_stake_owner = { save_temporary_scope_as = sm_bank }
			}
			else_if = {
				limit = { exists = var:sm_commerce_stake_owner }
				var:sm_commerce_stake_owner = { save_temporary_scope_as = sm_bank }
			}

			# At war with the bank: wartime seizure, ~25%/month.
			if = {
				limit = { is_at_war_with = scope:sm_bank }
				random_list = {
					25 = { sm_nationalize_bank_stakes = yes }
					75 = { }
				}
			}
			# Loathing plus desperation (broke or unstable): ~8%/month.
			else_if = {
				limit = {
					NOT = { opinion = { target = scope:sm_bank value >= -100 } }
					OR = {
						gold < 0
						stability < 0
					}
				}
				random_list = {
					8 = { sm_nationalize_bank_stakes = yes }
					92 = { }
				}
			}
			# Plain hostility: ~4%/month.
			else_if = {
				limit = { NOT = { opinion = { target = scope:sm_bank value >= -50 } } }
				random_list = {
					4 = { sm_nationalize_bank_stakes = yes }
					96 = { }
				}
			}
			# Broke and unfriendly: the treasury argues for seizure, ~4%/month.
			else_if = {
				limit = {
					gold < 0
					NOT = { opinion = { target = scope:sm_bank value >= 0 } }
				}
				random_list = {
					4 = { sm_nationalize_bank_stakes = yes }
					96 = { }
				}
			}
		}

		# --- Portfolio totals (this country as INVESTOR) for the ledger
		# header. Runs after the valuation caching so at least this
		# country's own targets-of-others are fresh; cross-country pulse
		# order makes some inputs up to a month stale, which is fine.
		sm_refresh_portfolio_totals = yes
	}
}
